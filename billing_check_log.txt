    import datetime
    from django.utils import timezone
    from core.models import TenantSubscription
    today = timezone.now().date()
    grace_deadline = today - datetime.timedelta(days=5)
    candidates = TenantSubscription.objects.select_related('tenant','plan').filter(
        status__in=['active','trial'],
        next_billing_date__isnull=False,
        next_billing_date__lte=grace_deadline,
    )
    marked = 0
    for sub in candidates:
        has_payment = sub.payments.filter(payment_date__gte=sub.next_billing_date).exists()
        if not has_payment:
            sub.status = 'past_due'
            sub.save(update_fields=['status','updated_at'])
            sub.sync_tenant_active()
            marked += 1
            print(f'VENCIDO: {sub.tenant.name} - next_billing_date={sub.next_billing_date}')
    total = TenantSubscription.objects.filter(status='past_due').count()
    print(f'Completado: {marked} nuevos vencimientos, {total} total vencidos')
